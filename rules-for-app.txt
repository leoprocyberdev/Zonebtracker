rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // This rule allows any user who has successfully signed in (request.auth != null),
    // including the Anonymous user created by your app, to read and write to 
    // ANY collection. This eliminates collection name typos as a source of error.
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
